SELECT 
	T1.STDNT_UNO
	, T1.FNM, T1.STDNT_NO
	, T2.CNO, T3.CLAS_NM
FROM
	TN_STDNT_INFO T1
	, TN_ENTSCH_INFO T2
	, TN_CLAS_INFO T3
WHERE T1.TNTSCH_UNO = T2.ENTSCH_UNO // TN_STDNT_INFO와 TN_ENTSCH_INFO의 교집합인 입교고유번호 칼럼 조건
AND T1.CLAS_UNO = T3.CLAS_UNO // TN_STDNT_INFO와 TN_CLAS_INFO의 교집합인 반고유번호 칼럼 조건
AND NVL(T1.DELETE_AT, 'N') = 'N' // NVL 함수 = 널처리 함수, data 값이 NULL일 떄 임의 설정값으로 처리해주기
// NVL(대상데이터, NULL일 때 데이터) => 값이 NULL 인 경우, 2번째 지정값을 출력하고 그렇지 않으면 대상데이터를 출력한다.

< 2중 서브쿼리 >
	(	
	SELECT * FROM (
		SELECT
			TVS.VCATN_SANCTN_UNO, TNS.VCATN_HIST_UNO, TVS.SANCTN_STTUS_CODE
			, ROW_NUMBER() OVER(PARITION BY TVS.VCATN_HIST_UNO ORDER BY TVS.REGIST_DT DESC) AS ROWINDEX
		FROM TH_VCATN_SANCTN TVS
	) WHERE ROWINDEX = 1
) C
